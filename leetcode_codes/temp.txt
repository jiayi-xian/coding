def numDistinct(s: str, t: str) -> int:   
    
    ns, nt = len(s)+1, len(t)+1
    dp = [ [0 for _ in range(nt) ] for _ in range(ns)]
    # dp[i][j] : ways tp generate string t[:j] from s[:i]

    for i in range(ns):
        dp[i][0] = 1 # non empty string s[:i] generate empty string t[:0]
    
    for i in range(1, ns):
        for j in range(1, nt):
            dp[i][j] = dp[i-1][j] + dp[i-1][j-1]*(s[i-1]==t[j-1])
    
    return dp[ns][nt]
